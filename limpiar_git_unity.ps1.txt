# Paso 1: Eliminar archivos rastreados que deberían ser ignorados
git rm -r --cached .

# Paso 2: Agregar nuevamente los archivos válidos (respetando .gitignore)
git add .

# Paso 3: Hacer commit de la limpieza
git commit -m "Limpieza final automática: eliminación de archivos ignorados (Library, Temp, etc.)"

# Paso 4: Push forzado al repositorio remoto
git push -u origin main --force

Write-Host "`n✅ Limpieza completada y push exitoso (si no hay errores)." -ForegroundColor Green
